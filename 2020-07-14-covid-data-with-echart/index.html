# 按需加载省市地图

现在有一个需求, 需要根据用户所在的省份来展示省市的疫情数据

显然, 全部加载所有的省级地图是不合理的, 需要按需加载

有试过其他的方案, 如 webpack 的 require.ensure, React Lazy + context 的形式, 但是都有问题

最终使用 script 标签来进行动态加载

let [option, setOption] = useState({});
useEffect(() => {
  showProvince(props.province);
  function showProvince(province) {
    loadMapScript(province, async function () {
      try {
        let data = await api.getProvinceData(provinceMap[province]);
        options.title.text = provinceMap[province] + '疫情地图';
        options.series[0].data = data;
        options.series[0].mapType = provinceMap[province];
        setOption(options);
      } catch (err) {
        message.error(err);
      }
    });
  }
  function loadMapScript(province, callback) {
    let s = document.createElement('script');
    s.setAttribute('src', `echarts/map/js/province/${province}.js`);
    s.onload = callback;
    document.body.appendChild(s);
  }
}, [option, props.province]);
